grulp es programación

Hay que instalar 3 plugins, gulp-uglify, gulp-concat and gulp-sourcemaps 

						npm install gulp-concat
						npm install gulp-sourcemaps
						npm install gulp-uglify
					

const gulp = require('gulp');
const uglify = require('gulp-uglify');
const concat = require('gulp-concat');
const sourcemaps 	= require('gulp-sourcemaps');

var configUglify = {
	mangle: false,
	compress: {
		sequences: true,
		properties: true,
		dead_code: true,
		drop_debugger: true,
		drop_console: true,
		conditionals: true,
		comparisons: true,
		booleans: true,
		if_return: true,
		join_vars: true,
		warnings: true
		
	},
	outSourceMap: true
};

var configMap = {
	loadMaps: false,
    debug: true
};

var configConcat = {
	newLine:'\n;'
};      

gulp.task('default', () => {
	return gulp
		.src('src/js/**/*.js')
		.pipe(plumber())
		.pipe(sourcemaps.init(configMap))
        .pipe(concat('app.js', configConcat))
		.pipe(uglify(configUglify))
       	.pipe(sourcemaps.write('.', {
            includeContent: true,
            sourceRoot: '/',
        }))
        .pipe(plumber.stop())
		.pipe(gulp.dest('dist/js/'));
});
