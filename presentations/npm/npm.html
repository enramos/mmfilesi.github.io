1. node, más que un servidor

  • desarrollar programas
  • compilar apps (nw)
  • gestionar programas (npm)

es una combinación de módulos

Un módulo es un fragmento de código encapsulado en una sola unidad

modulo-gato.js


var gato = {
  maullar: function() {
    console.log("Miau");
  },
  ronronear: function() {
    console.log("rrrrrr");
  }
}
module.exports = gato;

app.js


var moduloGato = require("./modulo-gato.js");
moduloGato.maullar();

Un ejemplo real


var express = require("express");
var app = express();

app.use(express.static('public'));
app.use(function(req, res) {
  res.status(404).send('404: ahí no hay ná de ná', 404);
});
app.use(function(err, req, res, next) {
  res.status(500).send('500: Sa roto!');
});
var server = app.listen(3000, function () {
  console.log('Servidor levantado en el puerto 3000');
});
